Python绘制饼状图示例

在这个示例中，我们将使用 Python 中的 matplotlib 库来绘制饼状图。
下面是一个简单的 Python 代码示例：

import matplotlib.pyplot as plt from matplotlib import font_manager # 数据 labels = ['Glare', 'Brightness', 'Denseness', 'Population', 'Development', 'Law', 'Climate', 'Geography', 'Biodiversity', 'Pollution'] sizes = [0.142857142857143, 0.285714285714286, 0.571428571428571, 0.285714285714286, 0.571428571428571, 0.142857142857143, 0.184607876049433, 0.125572837521839, 0.502291350087357, 0.187527936341371] # 分类 categories = ['Light', 'Society', 'Nature'] category_colors = {'Light': 'gold', 'Society': 'lightcoral', 'Nature': 'lightblue'} # 指标颜色 indicator_colors = [category_colors['Light'], category_colors['Light'], category_colors['Light'], category_colors['Society'], category_colors['Society'], category_colors['Society'], category_colors['Nature'], category_colors['Nature'], category_colors['Nature'], category_colors['Nature']] # 使用微软雅黑字体 font_path = font_manager.findSystemFonts(fontpaths=None, fontext='ttf')[0] font_prop = font_manager.FontProperties(fname=font_path, size=10) # 饼状图 fig, ax = plt.subplots(figsize=(10, 10)) wedges, texts, autotexts = ax.pie(sizes, labels=labels, autopct='%1.1f%%', startangle=90, colors=indicator_colors, wedgeprops=dict(width=0.4, edgecolor='w'), textprops=dict(fontproperties=font_prop)) # 添加数字标签在上方 for text, autotext in zip(texts, autotexts): text.set(size=10, weight='bold', color='black') # 标签颜色为黑色 autotext.set(size=10, weight='bold', va='center') # 添加标题 plt.title('Importance of Indicators', fontsize=16, fontweight='bold') # 添加图例 legend_labels = [plt.Line2D([0], [0], marker='o', color='w', markerfacecolor=category_colors[cat], markersize=10, label=cat) for cat in categories] plt.legend(handles=legend_labels, labels=categories, loc='upper left', bbox_to_anchor=(1, 1), fontsize=10) # 设置坐标轴刻度为空 plt.xticks([]) plt.yticks([]) # 调整布局 plt.tight_layout() # 显示图表 plt.show()
